province_event = {			#STABILITY GOES UP

	desc = EVTDESC1972015

			factor = 3
			has_game_rule = { name = stability_system value = reduced } 
		}

			factor = 0.9
			owner = {
				any_courtier_or_vassal = {
					has_character_modifier = ss_overseer
				}
			}
		}
		modifier = {
			factor = 0.7
			owner = { ai = yes }
		}
		modifier = {
			factor = 0.7
			owner = { trait = genius }
		}	
		modifier = {
			factor = 0.8
			owner = { trait = quick }
		}
		modifier = {
			factor = 0.9
			owner = { trait = bright }
		}		
		modifier = {
			factor = 1.25
			NOR = {
				has_province_modifier = instability_2
				has_province_modifier = instability_1
				has_province_modifier = stability_2
				has_province_modifier = stability_1
			}
		}	
		modifier = {
			factor = 0.70
			owner = { ai = yes }
			has_province_modifier = instability_2
		}
		modifier = {
			factor = 0.95
			owner = { ai = yes }
			has_province_modifier = instability_1
		}
		
		modifier = {
			factor = 0.9
			owner = { realm_diplomacy = 35 }
		}
		
		modifier = {
			factor = 0.9
			owner = { realm_diplomacy = 40 }
		}

		modifier = {
			factor = 0.8
			owner = { realm_diplomacy = 45 }
		}

		modifier = {
			factor = 2
			NOR = { 
				has_province_modifier = stability_1
				has_province_modifier = instability_1
				has_province_modifier = instability_2
			}
		}		

	}
	
	immediate = {
		
	}

		name = "EVTOPTA1972015"

		if = {
			limit = {
				has_province_modifier = stability_1

				name = stability_2

			}
		}

					has_province_modifier = stability_2

					has_province_modifier = instability_2

				name = stability_1
